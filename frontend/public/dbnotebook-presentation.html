Simple Integration

Query your documents programmatically with our REST API.

dbnotebook_client.py
import requests class DBNotebookClient: def __init__(self, base_url, api_key=None): self.base_url = base_url self.headers = {"X-API-Key": api_key} if api_key else {} self.conversation_id = None def query(self, notebook_id, query, mode="chat"): """Query documents with optional conversation memory.""" payload = { "notebook_id": notebook_id, "query": query, "mode": mode, "include_sources": True, "response_format": "detailed" # analytical | brief | default } if self.conversation_id: payload["conversation_id"] = self.conversation_id resp = requests.post( f"{self.base_url}/api/query", json=payload, headers=self.headers ) data = resp.json() self.conversation_id = data.get("conversation_id") return data # Usage client = DBNotebookClient("http://localhost:7860", "your-api-key") result = client.query("notebook-uuid", "What are the key findings?") print(result["answer"])
dbnotebook-client.js
class DBNotebookClient { constructor(baseUrl, apiKey = null) { this.baseUrl = baseUrl; this.headers = apiKey ? { 'X-API-Key': apiKey } : {}; this.conversationId = null; } async query(notebookId, query, mode = 'chat') { const payload = { notebook_id: notebookId, query: query, mode: mode, include_sources: true, response_format: 'detailed' }; if (this.conversationId) { payload.conversation_id = this.conversationId; } const response = await fetch(`${this.baseUrl}/api/query`, { method: 'POST', headers: { 'Content-Type': 'application/json', ...this.headers }, body: JSON.stringify(payload) }); const data = await response.json(); this.conversationId = data.conversation_id; return data; } } // Usage const client = new DBNotebookClient('http://localhost:7860', 'your-api-key'); const result = await client.query('notebook-uuid', 'What are the key findings?'); console.log(result.answer);
DBNotebookClient.java
import java.net.http.*; import com.google.gson.*; public class DBNotebookClient { private final String baseUrl; private final String apiKey; private String conversationId; public DBNotebookClient(String baseUrl, String apiKey) { this.baseUrl = baseUrl; this.apiKey = apiKey; } public JsonObject query(String notebookId, String query) { JsonObject payload = new JsonObject(); payload.addProperty("notebook_id", notebookId); payload.addProperty("query", query); payload.addProperty("mode", "chat"); payload.addProperty("include_sources", true); if (conversationId != null) { payload.addProperty("conversation_id", conversationId); } HttpRequest request = HttpRequest.newBuilder() .uri(URI.create(baseUrl + "/api/query")) .header("Content-Type", "application/json") .header("X-API-Key", apiKey) .POST(HttpRequest.BodyPublishers.ofString(payload.toString())) .build(); HttpResponse<String> response = HttpClient.newHttpClient() .send(request, HttpResponse.BodyHandlers.ofString()); JsonObject data = JsonParser.parseString(response.body()).getAsJsonObject(); conversationId = data.get("conversation_id").getAsString(); return data; } }

# Single query (stateless) curl -X POST http://localhost:7860/api/query \ -H "Content-Type: application/json" \ -H "X-API-Key: your-api-key" \ -d '{ "notebook_id": "your-notebook-uuid", "query": "What are the key findings?", "mode": "chat", "include_sources": true, "response_format": "detailed" }' # Conversational query (with memory) curl -X POST http://localhost:7860/api/query \ -H "Content-Type: application/json" \ -H "X-API-Key: your-api-key" \ -d '{ "notebook_id": "your-notebook-uuid", "query": "Tell me more about the second point", "conversation_id": "conv-uuid-from-previous", "mode": "chat" }' # List available notebooks curl http://localhost:7860/api/query/notebooks \ -H "X-API-Key: your-api-key"
